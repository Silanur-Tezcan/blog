💻 HTML Yapısı

            <div class="timer-container">
                <h2 class="text-warning">⏳ JavaScript Sayaç</h2>
                <p class="timer" id="display">00:00</p>
                
                <input type="number" id="minutes" placeholder="Dakika" class="form-control">
                <input type="number" id="seconds" placeholder="Saniye" class="form-control mt-2">

🚀 JavaScript ile Sayaç Mantığı

            let timer;
            let totalSeconds;

            function startTimer() {
                let minutes = document.getElementById("minutes").value || 0;
                let seconds = document.getElementById("seconds").value || 0;
                totalSeconds = parseInt(minutes) * 60 + parseInt(seconds);

                clearInterval(timer);
                timer = setInterval(function () {
                    if (totalSeconds > 0) {
                        totalSeconds--;
                        let min = Math.floor(totalSeconds / 60);
                        let sec = totalSeconds % 60;
                        document.getElementById("display").innerText = 
                            `${String(min).padStart(2, '0')}:${String(sec).padStart(2, '0')}`;
                    } else {
                        clearInterval(timer);
                        alert("⏳ Süre doldu!");
                    }
                }, 1000);
            }

            function stopTimer() {
                clearInterval(timer);
            }

            function resetTimer() {
                clearInterval(timer);
                document.getElementById("display").innerText = "00:00";
            }
